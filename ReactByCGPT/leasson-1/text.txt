ğŸ§ª React Lesson 6: Controlled Components (Forms)
âœ… What Is a Controlled Component?
In React, form inputs are controlled by React state using useState().

That means:
What you type into an input box is synced with a variable in state.

âœ… Step 1: Create FormInput.jsx
ğŸ“ src/FormInput.jsx

import { useState } from "react";

function FormInput() {
  const [name, setName] = useState("");

  const handleSubmit = (e) => {
    e.preventDefault(); // prevent form reload
    alert(`Hello, ${name}!`);
    setName(""); // clear input
  };

  return (
    <form onSubmit={handleSubmit}>
      <h3>ğŸ‘¤ Enter Your Name</h3>
      <input
        type="text"
        value={name}
        placeholder="Your name"
        onChange={(e) => setName(e.target.value)}
        style={{ padding: "8px", fontSize: "16px" }}
      />
      <button type="submit" style={{ marginLeft: "10px" }}>Submit</button>
      <p>Typed: {name}</p>
    </form>
  );
}

export default FormInput;
